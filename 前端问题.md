## 二、前端必须修改的问题

### 2.1 环境变量配置问题

**问题**：`.env` 中的 `VITE_API_BASE_URL` 配置不正确

**当前错误的配置**：
```env
VITE_API_BASE_URL=http://localhost:8080/v1/api
```

**原因**：健康检查接口 `/health` 不在 `/v1/api` 前缀下，应该单独处理

**正确配置**：
```env
VITE_API_BASE_URL=http://localhost:8080
VITE_API_PREFIX=/v1/api
```

**前端代码中需要正确拼接 URL**：
```javascript
// 健康检查（不走 prefix）
const healthUrl = `${VITE_API_BASE_URL}/health`;

// 其他 API（走 prefix）
const loginUrl = `${VITE_API_BASE_URL}${VITE_API_PREFIX}/auth/login`;
const profileUrl = `${VITE_API_BASE_URL}${VITE_API_PREFIX}/user/profile`;
```

### 2.2 user.id 类型问题

**问题**：前端可能将 user.id 当作数字处理

**后端实际返回**：
```json
{
  "id": "mqc8d5b497u5",
  "nickname": "小明",
  "email": "user@example.com"
}
```

**说明**：
- `user.id` 是**字符串**类型，不是数字
- 格式示例：`"mqc8d5b497u5"`、`"mqkyeg51dhdy"`
- 前端拼接字符串时不要用 `Number()` 转换

**前端代码示例**：
```javascript
// ✅ 正确
localStorage.setItem('userId', user.id); // id 是字符串

// ❌ 错误
const displayId = 'ID:' + Number(user.id); // 不要转换
```

## 三、前端代码检查清单

### 3.1 API 请求封装

```javascript
class ApiClient {
  constructor() {
    this.baseUrl = import.meta.env.VITE_API_BASE_URL; // http://localhost:8080
    this.prefix = import.meta.env.VITE_API_PREFIX;     // /v1/api
    this.token = localStorage.getItem('token');
  }

  // 获取完整 URL
  getUrl(path) {
    // 如果路径以 /health 开头，不加 prefix
    if (path === '/health') {
      return `${this.baseUrl}${path}`;
    }
    return `${this.baseUrl}${this.prefix}${path}`;
  }

  // 设置 Token
  setToken(token) {
    this.token = token;
    localStorage.setItem('token', token);
  }

  // 清除 Token
  clearToken() {
    this.token = null;
    localStorage.removeItem('token');
    localStorage.removeItem('user');
  }

  // 通用请求
  async request(method, path, data = null) {
    const url = this.getUrl(path);
    const headers = {
      'Content-Type': 'application/json'
    };

    if (this.token) {
      headers['Authorization'] = `Bearer ${this.token}`;
    }

    const options = { method, headers };
    if (data) {
      options.body = JSON.stringify(data);
    }

    try {
      const response = await fetch(url, options);
      const result = await response.json();

      if (!response.ok) {
        throw new Error(result.error || `HTTP ${response.status}`);
      }

      return result;
    } catch (error) {
      // 401 处理
      if (error.message.includes('Unauthorized') || 
          error.message.includes('401')) {
        this.clearToken();
        window.location.href = '/login';
      }
      throw error;
    }
  }

  // 快捷方法
  get(path) { return this.request('GET', path); }
  post(path, data) { return this.request('POST', path, data); }
  put(path, data) { return this.request('PUT', path, data); }
  delete(path) { return this.request('DELETE', path); }
}

export const api = new ApiClient();
```

### 3.2 登录页面

```javascript
async function handleLogin(email, password) {
  try {
    const result = await api.post('/auth/login', { email, password });
    
    // 保存登录状态
    api.setToken(result.token);
    localStorage.setItem('user', JSON.stringify(result.user));
    localStorage.setItem('userId', result.user.id); // 注意：id 是字符串
    
    // 跳转
    window.location.href = '/home';
    
    return { success: true };
  } catch (error) {
    // 显示中文错误
    return { success: false, message: error.message };
  }
}
```

## 七、常见错误排查

| 错误 | 原因 | 解决方法 |
|------|------|----------|
| `Failed to fetch` | 后端未启动 | 启动后端 `node src/index.js` |
| CORS 错误 | 后端未配置 | 已修复，确认前端请求正常 |
| `401 Unauthorized` | Token 过期/无效 | 清除 localStorage，重新登录 |
| `404 Not Found` | URL 拼接错误 | 检查 `VITE_API_BASE_URL` 配置 |
